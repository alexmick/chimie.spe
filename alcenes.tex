\textbf{Catalyse hétérogène} en plusieurs étapes dans le Nickel
\begin{enumerate}
	\item \textbf{Diffusion externe} des réactifs à la surface d'un grain de catalyseur
	\item \textbf{Diffusion interne} à l'intérieur des pores du grain
	\item \textbf{Adsorption} des réactifs
	\item \textbf{Réaction} entre les espèces
	\item \textbf{Désorption} des produits
	\item \textbf{Diffusion interne} des produits
	\item \textbf{Diffusion externe} des produits
\end{enumerate}
