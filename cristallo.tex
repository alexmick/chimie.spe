\section{Assemblages ioniques}
\begin{enumerate}
	\item Type NaCl: Cl$^{\scriptsize\ominus}$ noeuds du réseau, Na$^{\scriptsize\oplus}$ sites octaédriques \\
	Coordinence: [6;6] Population: $n_+=n_-=4$
	\item Type Blende (ZnS): S$^{2\scriptsize\ominus}$ noeuds du réseau, Zn$^{2\scriptsize\oplus}$ la moitié de sites tétraédriques (par bases alternées) \\
	Coordinence: [4;4] Population: $n_+=n_-=4$
\end{enumerate}
