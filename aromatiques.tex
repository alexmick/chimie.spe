\chapter{Hydrocarbures aromatiques}
\section{Halogénation}
\noindent \textbf{Bilan:} Ph - H + X$_2 \longrightarrow$ Ph - X + H$\,$X  \\
\textbf{Catalyseurs} selon la nature de X$_2$:
\begin{center}
\begin{description}
	\item[Br:] FeBr$_3$ généré \textit{in situ} par action de Br$_2$ sur Fe ( 2$\:$Fe + 3$\:$Br$_2 \longrightarrow\ 2\:$FeBr$_3$)
	\item[Cl:] AlCL$_3$ ou FeCl$_3$
	\item[I:] Trop mauvais rendement
	\item[F:] Trop explosif
\end{description}
\end{center}
